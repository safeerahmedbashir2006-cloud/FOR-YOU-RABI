    function goToScene(idOrNumber){
      document.querySelectorAll(".scene").forEach(scene => scene.classList.remove("active"));
      const targetId = typeof idOrNumber === "number" ? "scene" + idOrNumber : idOrNumber;
      const target = document.getElementById(targetId);
      if(target){
        target.classList.add("active");
        window.scrollTo({top:0,behavior:"smooth"});
      }
    }

    function openGiftMagic(){
      const box = document.getElementById("giftBox");
      if(box.classList.contains("opened")) return;
      box.classList.add("opened");
      setTimeout(() => goToScene(3), 1400);
    }

    function saveAnswer(){
      const textArea = document.getElementById("giftText");
      const value = textArea.value.trim();
      const status = document.getElementById("scene3Status");

      if(!value){
        status.textContent = "Please write something first, princess. 😏";
        status.className = "status error";
        return;
      }

      document.getElementById("savedAnswerPreview").textContent = value;
      status.textContent = "Saved successfully. 💖";
      status.className = "status success";

      setTimeout(() => goToScene(4), 500);
    }

    function checkPassword(){
      const value = document.getElementById("passwordInput").value;
      const status = document.getElementById("passwordStatus");

      if(value === correctPassword){
        status.textContent = "Correct password... unlocking now. ✨";
        status.className = "status success";
        setTimeout(() => goToScene(6), 650);
      } else {
        status.textContent = "Wrong password, pretty girl... try again. 💔";
        status.className = "status error";
      }
    }

    function startDaySequence(){
      goToScene("sceneWake");
      runBlinkAnimation();
      revealWakeupLines();
    }

    function runBlinkAnimation(){
      const overlay = document.getElementById("blinkOverlay");
      const pattern = [120,170,120,170,120,170];
      let total = 0;

      pattern.forEach((duration,index) => {
        setTimeout(() => {
          overlay.style.opacity = index % 2 === 0 ? "1" : "0";
        }, total);
        total += duration;
      });

      setTimeout(() => {
        overlay.style.opacity = "0";
      }, total + 60);
    }

    function revealWakeupLines(){
      const lines = document.querySelectorAll("#wakeLines .wake-line");
      lines.forEach(line => line.classList.remove("show"));

      lines.forEach((line,index) => {
        setTimeout(() => line.classList.add("show"), 800 + (index * 800));
      });

      setTimeout(() => {
        goToScene("scene9");
      }, 800 + (lines.length * 800) + 900);
    }

      goToScene("scene9Period");

    function setupDragHeart(){
      const dragHeart = document.getElementById("dragHeart");
      const fixedHeart = document.getElementById("fixedHeart");
      const area = document.getElementById("kissArea");
      const message = document.getElementById("kissMessage");

      let dragging = false;
      let offsetX = 0;
      let offsetY = 0;

      function getClientPosition(e){
        if(e.touches && e.touches[0]){
          return {x:e.touches[0].clientX, y:e.touches[0].clientY};
        }
        return {x:e.clientX, y:e.clientY};
      }

      function startDrag(e){
        if(kissDone) return;
        dragging = true;
        const pos = getClientPosition(e);
        const rect = dragHeart.getBoundingClientRect();
        offsetX = pos.x - rect.left;
        offsetY = pos.y - rect.top;
      }

      function doDrag(e){
        if(!dragging || kissDone) return;
        e.preventDefault();

        const pos = getClientPosition(e);
        const areaRect = area.getBoundingClientRect();

        let x = pos.x - areaRect.left - offsetX;
        let y = pos.y - areaRect.top - offsetY;

        x = Math.max(0, Math.min(x, areaRect.width - dragHeart.offsetWidth));
        y = Math.max(0, Math.min(y, areaRect.height - dragHeart.offsetHeight));

        dragHeart.style.left = x + "px";
        dragHeart.style.top = y + "px";

        const dragRect = dragHeart.getBoundingClientRect();
        const fixedRect = fixedHeart.getBoundingClientRect();

        const dragCenterX = dragRect.left + dragRect.width / 2;
        const dragCenterY = dragRect.top + dragRect.height / 2;
        const fixedCenterX = fixedRect.left + fixedRect.width / 2;
        const fixedCenterY = fixedRect.top + fixedRect.height / 2;

        const distance = Math.hypot(dragCenterX - fixedCenterX, dragCenterY - fixedCenterY);

        if(distance < 65){
          completeKissMission();
          dragging = false;
        }
      }

      function stopDrag(){
        dragging = false;
      }

      dragHeart.addEventListener("mousedown", startDrag);
      dragHeart.addEventListener("touchstart", startDrag, {passive:true});
      window.addEventListener("mousemove", doDrag);
      window.addEventListener("touchmove", doDrag, {passive:false});
      window.addEventListener("mouseup", stopDrag);
      window.addEventListener("touchend", stopDrag);

      message.textContent = "Drag the left heart to Safeer’s heart 😏";
    }

    function completeKissMission(){
      if(kissDone) return;
      kissDone = true;

      const burst = document.getElementById("kissBurst");
      const message = document.getElementById("kissMessage");
      const dragHeart = document.getElementById("dragHeart");

      dragHeart.style.opacity = "0";
      burst.innerHTML = "";
      burst.classList.add("show");

      for(let i=0; i<12; i++){
        const heart = document.createElement("div");
        heart.className = "mini-heart";
        heart.textContent = ["💖","💕","💋"][Math.floor(Math.random() * 3)];
        heart.style.left = (45 + Math.random() * 20) + "%";
        heart.style.top = (40 + Math.random() * 18) + "%";
        heart.style.setProperty("--x", ((Math.random() * 100) - 50) + "px");
        heart.style.animationDelay = (Math.random() * 0.15) + "s";
        burst.appendChild(heart);
      }

      message.textContent = "Kiss delivered... Safeer is smiling now 😌";

      setTimeout(() => {
        markMissionComplete(1);
        backToOverview();
      }, 1400);
    }
